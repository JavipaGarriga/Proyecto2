CREATE OR REPLACE TABLE proyecto-470220.datos.artistas_aux AS
WITH base AS (
-- Tomamos datos principales desde tabla_unificada (que ya trae streams y playlists)
SELECT
u.track_id,
u.artist_count,
u.streams_limpio,
u.playlists_totales,
s.artist_name
FROM proyecto-470220.datos.tabla_unificada u
LEFT JOIN proyecto-470220.datos.spotify s
ON u.track_id = s.track_id
WHERE s.artist_name IS NOT NULL
),
normalizado AS (
-- Unificamos todo tipo de separadores a "|"
SELECT
track_id,
artist_count,
streams_limpio,
playlists_totales,
REGEXP_REPLACE(
artist_name,
r'(?i)\s*(feat\.?|featuring|with|&|and|,|\/|\+|\bx\b|\b y \b|\bcon\b|vs\.?)\s*',
'|'
) AS artists_pipe
FROM base
),
explotado AS (
-- Una fila por artista
SELECT
track_id,
artist_count,
streams_limpio,
playlists_totales,
TRIM(artist) AS artista
FROM normalizado,
UNNEST(SPLIT(artists_pipe, '|')) AS artist
WHERE artist IS NOT NULL AND artist != ''
)
SELECT
artista,
COUNT(DISTINCT track_id) AS total_canciones,
COUNTIF(artist_count > 1) AS total_colaboraciones,
SUM(playlists_totales) AS playlists_totales,
SUM(streams_limpio) AS streams_totales
FROM explotado
GROUP BY artista
ORDER BY total_canciones DESC, total_colaboraciones DESC;
